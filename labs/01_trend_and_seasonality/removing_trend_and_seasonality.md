# Remove trend using first differences

component.

    for (i in 1:length(data$GVA_no_trend)) {
      if (i == 1) {
        data$GVA_no_trend[i] <- NA
      } else {
        data$GVA_no_trend[i] <- data$GVA[i] - data$GVA[i - 1]
      }
    }
